    def __init__(self, **entries):
        # WE initialize by the default training params, overridden by the provided params
        super().__init__(**DEFAULT_TRAINING_PARAMS)
        self.set_schema(TRAINING_PARAM_SCHEMA)
        if len(entries) > 0:
            self.override(**entries)
